% 中英文摘要和关键字

\begin{abstract}
  % 关键词用“英文逗号”分隔，输出时会自动处理为正确的分隔符
  1
  \thusetup{
    keywords = {关键词 1, 关键词 2, 关键词 3, 关键词 4, 关键词 5},
  }
\end{abstract}

\begin{abstract*}
  % Use comma as separator when inputting
  2
  \thusetup{
    keywords* = {keyword 1, keyword 2, keyword 3, keyword 4, keyword 5},
  }
\end{abstract*}
